\begin{frame}{Background: Flow Matching \& Diffusion Transformers}
\textbf{Flow Matching / Interpolant Framework:}

Given clean data $x_0$ and noise $\epsilon \sim \mathcal{N}(0, I)$, construct noisy input:
\begin{equation}
  x_t = \alpha_t \, x_0 + \sigma_t \, \epsilon
\end{equation}

\textbf{Linear path:} $\alpha_t = 1 - t$, \; $\sigma_t = t$, \quad $t \in [0, 1]$

\vspace{0.5em}
\textbf{Velocity (v-prediction) target:}
\begin{equation}
  v_t = \dot{\alpha}_t \, x_0 + \dot{\sigma}_t \, \epsilon
\end{equation}

For linear path: $v_t = -x_0 + \epsilon$

\vspace{0.5em}
\textbf{Diffusion Transformers:}
\begin{itemize}
  \item \textbf{DiT} (Peebles \& Xie, 2023): Transformer backbone replaces U-Net
  \item \textbf{SiT} (Ma et al., 2024): DiT + Scalable Interpolant framework
  \item Patch embedding $\rightarrow$ Transformer blocks with adaLN-Zero $\rightarrow$ Unpatchify
\end{itemize}
\end{frame}

\section{Method: REPA}
% ############################################################

% ============================================================
% SLIDE 10: REPA Overview
% ============================================================
\begin{frame}{REPA: REPresentation Alignment}
\begin{columns}[T]
\begin{column}{0.54\textwidth}
\includegraphics[width=\linewidth,max height=0.78\textheight]{feat_matching_fig.png}
\end{column}
\begin{column}{0.44\textwidth}
\small
\vspace{1em}
\textbf{Core idea:}\\[4pt]
Align the hidden representation of the denoising network (at an intermediate layer $l$) with the clean image representation from a frozen pretrained encoder.

\vspace{0.8em}
\textbf{Key components:}
\begin{itemize}\setlength{\itemsep}{3pt}
  \item Frozen SSL encoder (DINOv2)
  \item Projector MLP on hidden states
  \item Cosine similarity alignment loss
  \item No overhead at inference
\end{itemize}
\end{column}
\end{columns}
\end{frame}

% ============================================================
% SLIDE 11: Flow Matching Formulation
% ============================================================
\begin{frame}{Flow Matching Formulation}
\textbf{Stochastic interpolant:}
\begin{equation}
  x_t = \alpha_t \, x_0 + \sigma_t \, \epsilon, \quad \epsilon \sim \mathcal{N}(0, I), \quad t \in [0, 1]
\end{equation}

\textbf{Path types:}
\begin{itemize}
  \item \textbf{Linear:} $\alpha_t = 1 - t, \quad \sigma_t = t$
  \item \textbf{Cosine:} $\alpha_t = \cos\!\left(\frac{\pi t}{2}\right), \quad \sigma_t = \sin\!\left(\frac{\pi t}{2}\right)$
\end{itemize}

\vspace{0.5em}
\textbf{Velocity field (v-prediction):}
\begin{equation}
  v_t = \dot{\alpha}_t \, x_0 + \dot{\sigma}_t \, \epsilon
\end{equation}

The model $v_\theta(x_t, t)$ is trained to predict this velocity.

\vspace{0.5em}
\textbf{Denoising loss:}
\begin{equation}
  \loss_{\text{denoise}} = \E_{t, x_0, \epsilon}\left[\left\| v_\theta(x_t, t) - v_t \right\|^2\right]
\end{equation}
\end{frame}

% ============================================================
% SLIDE 12: REPA Alignment Loss
% ============================================================
\begin{frame}{REPA Alignment Loss}
\textbf{Setup:} Given a frozen pretrained encoder $f_{\text{enc}}$ (e.g., DINOv2):

\vspace{0.3em}
\begin{enumerate}
  \item Extract \textbf{clean image features}: $z = f_{\text{enc}}(x_0) \in \R^{N \times d_z}$ \quad (patch tokens)
  \item Extract \textbf{noisy hidden states} at layer $l$: $h^{(l)} = \text{DiT-Block}_l(x_t, t, y)$
  \item Project hidden states: $\tilde{z} = f_{\text{proj}}(h^{(l)}) \in \R^{N \times d_z}$
\end{enumerate}

\vspace{0.5em}
\textbf{Alignment loss} (negative cosine similarity):
\begin{equation}
  \loss_{\text{align}} = - \E_{t, x_0, \epsilon}\left[
    \frac{1}{N} \sum_{j=1}^{N}
    \left\langle
      \frac{\tilde{z}_j}{\|\tilde{z}_j\|},\;
      \frac{z_j}{\|z_j\|}
    \right\rangle
  \right]
\end{equation}

where $N$ is the number of patch tokens, and the sum is over \textbf{spatial positions}.

\vspace{0.5em}
\textbf{Intuition:} Force the diffusion model's intermediate representation to ``see'' what DINOv2 sees in the clean image, even from noisy input.
\end{frame}

% ============================================================
% SLIDE 13: Total Objective
% ============================================================
\begin{frame}[fragile]{Total Training Objective}
\textbf{Combined loss:}
\begin{equation}
  \boxed{
    \loss = \loss_{\text{denoise}} + \lambda \cdot \loss_{\text{align}}
  }
\end{equation}

where $\lambda$ controls the alignment strength (default: $\lambda = 0.5$).

\vspace{0.6em}
\textbf{Projector MLP} $f_{\text{proj}}$: maps hidden states to encoder space

% ============================================================
% SLIDE 14: Design Choices
% ============================================================
\begin{frame}{Design Choices}
\textbf{Which layer to align?}
\begin{itemize}
  \item Alignment applied at layer $l$ = \texttt{encoder\_depth}
  \item Default: $l = 8$ for SiT-XL/2 (28 total blocks)
  \item \textbf{Early layers} $\Rightarrow$ aligns low-level + semantic features
\end{itemize}

\vspace{0.5em}
\textbf{Which encoder?}
\begin{itemize}
  \item \textbf{DINOv2} (ViT-B, ViT-L, ViT-g) -- best overall
  \item Also works with: CLIP, MoCo v3, I-JEPA, MAE, DINOv1
\end{itemize}

\vspace{0.5em}
\textbf{Key design principle:}
\begin{center}
\fbox{
  \parbox{0.85\textwidth}{
    \centering
    Align \textbf{early layers} with SSL encoder \\
    $\Rightarrow$ Later layers are freed to focus on \textbf{high-frequency generation details}
  }
}
\end{center}

\vspace{0.3em}
\textbf{Other details:}
\begin{itemize}
  \item Projector is \textit{only used during training} (discarded at inference)
  \item Encoder is \textit{frozen} -- no gradient flows back
  \item $\lambda = 0.5$, AdamW optimizer, lr = $10^{-4}$, batch size = 256
\end{itemize}
\end{frame}

% SLIDE 15: How REPA Works -- Intuition
% ============================================================
\begin{frame}{How REPA Works: Layer Specialization}
\begin{columns}[T]
\begin{column}{0.52\textwidth}
\small
\textbf{Without REPA (Vanilla):}
\begin{itemize}\setlength{\itemsep}{1pt}
  \item All layers must jointly learn both \textit{semantic understanding} and \textit{fine-grained generation}
  \item Slow convergence -- early layers struggle
\end{itemize}

\vspace{0.4em}
\textbf{With REPA:}
\begin{itemize}\setlength{\itemsep}{1pt}
  \item \textbf{Early layers} ($\leq l$): Aligned with SSL encoder $\Rightarrow$ semantics ``injected''
  \item \textbf{Later layers} ($> l$): Free for high-freq details
\end{itemize}
\end{column}

% ============================================================
% SLIDE 26: Key Takeaways
% ============================================================
\begin{frame}{Key Takeaways}
\begin{enumerate}
  \item \textbf{Representation quality is the bottleneck} in training diffusion transformers
  \vspace{0.3em}
  \item \textbf{REPA} is a simple regularization: align diffusion model hidden states with pretrained SSL encoders using cosine similarity
  \vspace{0.3em}
  \item Results are \textbf{striking}:
  \begin{itemize}
    \item \textbf{17.5$\times$} training speedup
    \item State-of-the-art \textbf{FID = 1.42} on ImageNet 256$\times$256
    \item Works with diverse encoders (DINOv2, CLIP, I-JEPA, MAE, ...)
  \end{itemize}
  \vspace{0.3em}
  \item REPA creates \textbf{layer specialization}: early layers for semantics, later layers for generation details
  \vspace{0.3em}
  \item \textbf{Practical takeaway:} Training powerful diffusion transformers is now \textit{much easier and cheaper}
\end{enumerate}
\end{frame}

% ============================================================
% SLIDE 27: Limitations & Future Work
% ============================================================
\begin{frame}{Limitations \& Future Work}
\textbf{Limitations:}
\begin{itemize}
  \item Requires access to a pretrained visual encoder (additional dependency)
  \item Currently demonstrated primarily on ImageNet -- scaling to diverse domains needs study
  \item The projector MLP adds a small overhead during training (discarded at inference)
\end{itemize}

\vspace{1em}
\textbf{Future directions:}
\begin{itemize}
  \item Apply REPA to \textbf{text-to-image} generation at scale (initial MS-COCO results shown)
  \item Explore alignment with \textbf{multi-modal encoders} (e.g., CLIP for text-guided alignment)
  \item Combine with \textbf{progressive training} or \textbf{distillation} for even faster convergence
  \item Investigate \textbf{adaptive alignment depth} -- dynamically choose layer $l$
\end{itemize}
\end{frame}

  \item Rule of thumb: align at $\sim$25--30\% of total depth
\end{itemize}

\begin{frame}{Ablation: Loss Function Variants}
\small
\textbf{What type of alignment loss works best?}

\vspace{0.5em}
\begin{center}
\begin{tabular}{lccc}
\toprule
\textbf{Loss Type} & \textbf{Formula} & \textbf{FID $\downarrow$} & \textbf{Linear Acc.} \\
\midrule
\textbf{Cosine sim.} (REPA) & $-\langle \bar{\tilde{z}}, \bar{z} \rangle$ & \textbf{23.8} & \textbf{66.1} \\
MSE & $\|\tilde{z} - z\|^2$ & 28.4 & 62.7 \\
Smooth L1 & smooth-$\ell_1$($\tilde{z}$, $z$) & 27.9 & 63.1 \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.5em}
\textbf{Why cosine similarity?}
\begin{itemize}\setlength{\itemsep}{2pt}
  \item Normalization makes the loss \textbf{scale-invariant}
  \item Focuses on \textbf{directional alignment}, not magnitude
  \item More stable training -- avoids magnitude mismatch between different encoder spaces
  \item Consistent with self-supervised learning objectives (e.g., DINO, BYOL)
\end{itemize}
\end{frame}
